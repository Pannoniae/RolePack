	id = gcg_nuclearstartup.1
	title = gcg_nuclearstartup.1.t
	desc = gcg_nuclearstartup.1.d
	is_triggered_only = yes

	
	option = {
		name = gcg_nuclearstartup.1.a
		set_global_flag = gcg_startup_event
		ROOT = { country_event = { days = 0 id = gcg_nuclearstartup.2 } }
	}
	
	option = {
		name = gcg_nuclearstartup.1.b
		set_global_flag = gcg_radiation_startup_event
		every_country = {
		set_technology = {
		gcg_nbc_minor_tech = 1
		} }
		set_global_flag = gcg_disable_unique_units
		every_country = { set_technology = { gcg_staff_company = 1 } }
		set_global_flag = no_staff_units
	}


}

country_event = {
	id = gcg_nuclearstartup.7
	title = gcg_nuclearstartup.7.t
	desc = gcg_nuclearstartup.7.d
	is_triggered_only = yes

	
	option = {
		name = gcg_nuclearstartup.7.a
		IF = { LIMIT = { GER = { has_government = fascism is_ai = no } } GER = { set_country_flag = gcg_germany_fascist set_technology = { gcg_ger_the_ss = 1 } load_oob = "gcg_ger_spawn_units" }  } 
		IF = { LIMIT = { GER = { has_government = fascism is_ai = yes } } GER = { set_country_flag = gcg_germany_fascist set_technology = { gcg_ger_the_ss = 1 } load_oob = "gcg_gerai_spawn_units" load_oob = "gcg_gerai_spawn_units" }  } 
		IF = { LIMIT = { SOV = { has_government = communism is_ai = no } } SOV = { set_country_flag = gcg_russia_communist } SOV = { set_technology = { gcg_sov_nkvd = 1 gcg_sov_guards_rifle = 1 } load_oob = "gcg_sov_spawn_units" } }
		IF = { LIMIT = { SOV = { has_government = communism is_ai = yes } } SOV = { set_country_flag = gcg_russia_communist } SOV = { set_technology = { gcg_sov_nkvd = 1 gcg_sov_guards_rifle = 1 } load_oob = "gcg_sovai_spawn_units" load_oob = "gcg_sovai_spawn_units" } }
		ITA = { load_oob = "gcg_ita_gas" }
		every_country = { set_technology = { gcg_staff_company = 1 } }
		ENG = { set_country_flag = gcg_is_united_kingdom }
		set_global_flag = gcg_enable_unique_units
		ROOT = { country_event = { days = 0 id = gcg_nuclearstartup.11 } }
	}
	
	option = {
		name = gcg_nuclearstartup.7.b
		set_global_flag = gcg_disable_unique_units
		every_country = { set_technology = { gcg_staff_company = 1 } }
		ROOT = { country_event = { days = 0 id = gcg_nuclearstartup.11 } }
		ITA = { load_oob = "gcg_ita_gas" }
	}


}
