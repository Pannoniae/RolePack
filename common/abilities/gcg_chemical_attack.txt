chlorine_attack = {
        name = "Chlorine Attack"
        desc = chlorine_ability_desc
        
        sound_effect = command_power_ability_special
        
		
		
        type = army_leader
        
        allowed = {
		NOT = { has_trait = gcg_cold_calculations 
		has_trait = gcg_using_phosgene_trait

		has_trait = gcg_using_vx_trait
		has_trait = gcg_using_phosgene_artillery_trait
		has_trait = gcg_using_sarin_trait
		}
			OWNER = {
				has_equipment = { gcg_chlorine_gas_eq > 49 }
				OR = {
				NOT = { any_enemy_country = { has_tech = gcg_gas_masks } }
				has_tech = gcg_mask_breakers
				}
            }
			
        }
		
		one_time_effect = {
			OWNER = {
				add_equipment_to_stockpile = { type = gcg_chlorine_gas_eq amount = -50  }
				if = { limit = { NOT = { has_global_flag = gcg_first_use_chemical_weapons } } set_global_flag = gcg_first_use_chemical_weapons news_event = { id = gcg_chemicals.1 days = 1 } }
				add_timed_idea = {
    			idea = gcg_recently_used_chemical_weapons
   				days = 21
				}
				}
				
				add_temporary_buff_to_units = {
				combat_breakthrough = -0.05 org_damage_multiplier = 0.10 combat_defense = 0.75 combat_offense = 0.50 
				days = 4
			}
			add_timed_unit_leader_trait = {
			trait = gcg_using_chlorine_trait
			days = 4
			}
        }
        
        ai_will_do = {
            factor = 0
			modifier = {
			factor = 10
			has_global_flag = gcg_first_use_chemical_weapons
			}
        }
        
        cost = 0.10
