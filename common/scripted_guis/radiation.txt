scripted_gui = {
	gcg_radiation = {
	
		context_type = selected_state_context
		window_name = "gcg_radiation_window"
		parent_window_token = selected_state_view
		visible = {
			check_variable = { ROOT.hide_radiation_tab = 0 }
		}
		
	}
	
	gcg_global_radiation = {
	
		context_type = player_context
		window_name = "gcg_background_radiation_window"
		parent_window_token = top_bar
		visible = {
			check_variable = { ROOT.hide_radiation_tab = 0 }
		}
	}
	
	gcg_hide_button = {
	
		context_type = player_context
		window_name = "gcg_hide_button_window"
		parent_window_token = top_bar
		visible = {
			always = yes
		}
		
		effects = {
			gcg_hide_radiation_button_click = {
			if = {
					limit = { check_variable = { ROOT.hide_radiation_tab = 0 } }
					set_variable = { ROOT.hide_radiation_tab = 1 }
				}
				else = {
				if = {
						limit = { check_variable = { ROOT.hide_radiation_tab > 0.9 } }
						set_variable = { ROOT.hide_radiation_tab = 0 }
					}
				}
			}
		}
	}
	
	gcg_global_death_count = {
	
		context_type = player_context
		window_name = "gcg_global_ded_window"
		parent_window_token = top_bar
		visible = {
			check_variable = { ROOT.hide_radiation_tab = 0 }
		}
	}
	
	gcg_state_radiation_casualties = {
	
		context_type = selected_state_context
		window_name = "gcg_state_radiation_casualties_window"
		parent_window_token = selected_state_view
		visible = {
			check_variable = { ROOT.hide_radiation_tab = 0 }
		}
	}
}
